1핵심 원칙

PRINCIPLE 01
단계 분리 (Phased)
한 마이그레이션은 독립적으로 실행·검증·롤백 가능해야 함. 한 단계 실패가 전체 시스템을 무너뜨리면 안 됨.
PRINCIPLE 02
기존 시스템 무중단
새 테이블만 추가, 기존 테이블은 마지막 단계 전까지 건드리지 않음. 비모 ERP는 마이그레이션 중에도 계속 동작.
PRINCIPLE 03
이중 검증 (Coexist)
기존 권한 체크 + 새 권한 체크 둘 다 통과해야 진입. 새 시스템 버그가 있어도 기존 시스템이 fallback 역할.
PRINCIPLE 04
멱등성 (Idempotent)
모든 SQL은 IF NOT EXISTS, ON CONFLICT DO NOTHING 등으로 재실행해도 안전해야 함.

4Phase 1 · 새 테이블 생성

P1

인증 코어 + 파트너 메타 테이블 생성

기존 시스템 영향 0% — 새 테이블만 추가하므로 안전한 첫 단계
위험 낮음 ⏱ 약 5분 ↺ DROP만
목표

CREATE TABLE IF NOT EXISTS public.apps (

  name        text NOT NULL,
  domain      text,
  sso_enabled boolean DEFAULT true,
  created_at  timestamptz DEFAULT now()

CREATE TABLE IF NOT EXISTS public.profiles (
  id          uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  user_type   text NOT NULL DEFAULT 'staff'
              CHECK (user_type IN ('staff', 'partner', 'external')),
  name        text,
  avatar_url  text,
  phone       text,
  created_at  timestamptz DEFAULT now(),
  updated_at  timestamptz DEFAULT now()
);

-- 권한 매핑 (N:N user × app)
CREATE TABLE IF NOT EXISTS public.app_access (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id          uuid NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
  app_code         text NOT NULL REFERENCES public.apps(code),
  role             text NOT NULL DEFAULT 'member',
  status           text NOT NULL DEFAULT 'active'
                   CHECK (status IN ('active', 'suspended')),
  joined_at        timestamptz DEFAULT now(),
  last_accessed_at timestamptz,
  UNIQUE (user_id, app_code)
);

-- 비모 팀원 상세
CREATE TABLE IF NOT EXISTS public.vimo_staff (
  profile_id uuid PRIMARY KEY REFERENCES public.profiles(id) ON DELETE CASCADE,
  department text,
  position   text,
  hire_date  date,
  created_at timestamptz DEFAULT now()
);

-- 파트너 ERP 사용자 상세 (이름 충돌 방지: 'partners' → 'partner_meta')
CREATE TABLE IF NOT EXISTS public.partner_meta (
  profile_id    uuid PRIMARY KEY REFERENCES public.profiles(id) ON DELETE CASCADE,
  type          text NOT NULL CHECK (type IN ('freelancer', 'business')),
  tier          text,
  bank_info     jsonb,
  work_formats  text[] DEFAULT '{}',
  status        text NOT NULL DEFAULT 'pending'
                CHECK (status IN ('pending', 'active', 'suspended')),
  started_at    date,
  legacy_partner_id uuid -- 기존 partners 테이블 매핑 (Phase 5에서 채움),
  created_at    timestamptz DEFAULT now()
);

-- 일회용 초대 토큰
CREATE TABLE IF NOT EXISTS public.partner_invites (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  token         text UNIQUE NOT NULL,
  profile_id    uuid REFERENCES public.profiles(id) ON DELETE CASCADE,
  invited_email text,
  invited_by    uuid REFERENCES public.profiles(id),
  expires_at    timestamptz NOT NULL DEFAULT (now() + INTERVAL '7 days'),
  used_at       timestamptz,
  status        text NOT NULL DEFAULT 'pending'
                CHECK (status IN ('pending', 'used', 'expired', 'revoked')),
  created_at    timestamptz DEFAULT now()
);

-- 인덱스

5Phase 2 · 제약·트리거·RLS

P2

상호 배제 + user_type 검증 + RLS 정책

새 테이블에 보호막 — 잘못된 데이터 INSERT 원천 차단
위험 낮음 ⏱ 약 5분 ↺ DROP만
SQL · 부분 유니크 인덱스 (상호 배제)
-- 한 user_id는 vimo_erp / partner_erp 중 단 하나만 가질 수 있음
CREATE UNIQUE INDEX IF NOT EXISTS app_access_erp_exclusive
ON public.app_access (user_id)
WHERE app_code IN ('vimo_erp', 'partner_erp');
SQL · 트리거 (user_type ↔ app_code 정합성)
CREATE OR REPLACE FUNCTION public.check_app_access_type()
RETURNS trigger AS $$
DECLARE
  v_user_type text;
BEGIN
  SELECT user_type INTO v_user_type
  FROM public.profiles WHERE id = NEW.user_id;

  IF NEW.app_code = 'vimo_erp' AND v_user_type != 'staff' THEN
    RAISE EXCEPTION 'vimo_erp는 staff만 접근 가능 (현재: %)', v_user_type;
  END IF;

  IF NEW.app_code = 'partner_erp' AND v_user_type != 'partner' THEN
    RAISE EXCEPTION 'partner_erp는 partner만 접근 가능 (현재: %)', v_user_type;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS enforce_app_access_type ON public.app_access;
CREATE TRIGGER enforce_app_access_type
  BEFORE INSERT OR UPDATE ON public.app_access
  FOR EACH ROW EXECUTE FUNCTION public.check_app_access_type();
SQL · RLS 정책 (핵심만)
-- profiles: 본인만 SELECT/UPDATE, vimo_erp admin은 전체 SELECT
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;

CREATE POLICY "본인 프로필 조회" ON public.profiles
  FOR SELECT USING (auth.uid() = id);

CREATE POLICY "본인 프로필 수정" ON public.profiles
  FOR UPDATE USING (auth.uid() = id);

CREATE POLICY "vimo_erp admin 전체 조회" ON public.profiles
  FOR SELECT USING (
    EXISTS (SELECT 1 FROM public.app_access
            WHERE user_id = auth.uid()
              AND app_code = 'vimo_erp'
              AND role = 'admin'
              AND status = 'active')
  );

-- app_access: 본인 권한 조회만, admin이 전체 관리
ALTER TABLE public.app_access ENABLE ROW LEVEL SECURITY;

CREATE POLICY "본인 권한 조회" ON public.app_access
  FOR SELECT USING (auth.uid() = user_id);

CREATE POLICY "vimo_erp admin 권한 관리" ON public.app_access
  FOR ALL USING (
    EXISTS (SELECT 1 FROM public.app_access aa
            WHERE aa.user_id = auth.uid()
              AND aa.app_code = 'vimo_erp'
              AND aa.role = 'admin'
              AND aa.status = 'active')
  );

-- partner_meta: 본인만 + admin
ALTER TABLE public.partner_meta ENABLE ROW LEVEL SECURITY;
CREATE POLICY "본인 메타 조회" ON public.partner_meta
  FOR SELECT USING (auth.uid() = profile_id);

-- partner_invites: 토큰으로만 접근 (anonymous 허용)
ALTER TABLE public.partner_invites ENABLE ROW LEVEL SECURITY;
CREATE POLICY "토큰으로 조회" ON public.partner_invites
  FOR SELECT USING (status = 'pending' AND expires_at > now());

6Phase 3 · 기존 사용자 백필

P3

user_profiles → profiles + app_access(vimo_erp) 자동 생성

기존 비모 ERP 사용자에게 새 권한 시스템 권한 부여
위험 중간 ⏱ 약 10분 ↺ DELETE FROM
전략
  • 모든 user_profiles row를 profiles에 복사 (user_type='staff')
  • 승인된 사용자 + admin → app_access(vimo_erp, status='active')
  • 미승인 사용자 → app_access(vimo_erp, status='suspended')
  • 비모 팀이 아닌 user (예: 테스트 계정)은 수동 분류 후 백필
SQL · profiles 백필
-- 멱등 — 이미 있는 id는 INSERT 안 됨
INSERT INTO public.profiles (id, user_type, name, created_at)
SELECT
  up.id,
  'staff' AS user_type,                -- 모두 staff로 시작
  COALESCE(au.raw_user_meta_data->>'name', au.email),
  COALESCE(up.created_at, now())
FROM public.user_profiles up
JOIN auth.users au ON au.id = up.id
ON CONFLICT (id) DO NOTHING;

-- vimo_staff 메타 백필
INSERT INTO public.vimo_staff (profile_id, position)
SELECT id, 'manager' FROM public.profiles
ON CONFLICT (profile_id) DO NOTHING;
SQL · app_access 백필
-- vimo_erp 권한 부여
INSERT INTO public.app_access (user_id, app_code, role, status)
SELECT
  up.id,
  'vimo_erp',
  COALESCE(up.role, 'member'),
  CASE
    WHEN up.approved = true OR up.role = 'admin' THEN 'active'
    ELSE 'suspended'
  END
FROM public.user_profiles up
ON CONFLICT (user_id, app_code) DO NOTHING;

롤백
-- 백필한 row만 정확히 제거 (user_profiles에 있던 것만)
DELETE FROM public.app_access
WHERE app_code = 'vimo_erp'
  AND user_id IN (SELECT id FROM public.user_profiles);

DELETE FROM public.vimo_staff
WHERE profile_id IN (SELECT id FROM public.user_profiles);

DELETE FROM public.profiles
WHERE id IN (SELECT id FROM public.user_profiles);

7Phase 4 · 비모 ERP 이중 검증

P4

코드 변경 — 두 검증 모두 통과해야 입장. 회귀 테스트 핵심.

코드 변경 — video-moment/src/proxy.ts
// 기존
const { data: profile } = await supabase
  .from('user_profiles')
  .select('approved, role, needs_password_change')
  .eq('id', user.id)
  .single();

if (!profile || (profile.role !== 'admin' && profile.approved !== true)) { ... }

// 변경 — 이중 검증 추가
const [{ data: profile }, { data: access }] = await Promise.all([
  supabase.from('user_profiles')
    .select('approved, role, needs_password_change')
    .eq('id', user.id).single(),
  supabase.from('app_access')
    .select('status')
    .eq('user_id', user.id)
    .eq('app_code', 'vimo_erp')
    .single(),
]);

const profileOk = profile && (profile.role === 'admin' || profile.approved === true);
const accessOk = access && access.status === 'active';

// 둘 다 통과해야 진입
if (!profileOk || !accessOk) {
  await supabase.auth.signOut();
  return NextResponse.redirect(loginUrl);
}
테스트 시나리오
  1. 정상 비모 팀원 → 로그인 → 대시보드 진입 ✓
  2. 승인된 admin → 로그인 → 대시보드 진입 ✓
  3. app_access 없는 사용자 임의 생성 → 로그인 → 거부 ✓
  4. app_access.status='suspended' 변경 → 즉시 로그아웃 ✓
  5. 주요 페이지 회귀: 매니지먼트, 프로젝트, 정산, 클라이언트, 파트너, 메일
롤백

8Phase 5 · 파트너 데이터 정리

P5

비즈니스 결정 + 데이터 백업 필수 단계

전제 조건

-- ⚠ 이 방식은 가입 후 id 교체 필요. 더 나은 방식은 가입 시점에 동적 생성

-- 권장 방식: 빈 profile 생성 안 하고, 가입(invite token 사용) 시
-- auth.users.id를 받아 profile + partner_meta 동시 생성

-- partner_meta 메타 데이터 일괄 등록
INSERT INTO public.partner_meta (profile_id, type, tier, started_at,
                                  status, legacy_partner_id)
SELECT
  pr.id,
  CASE WHEN p.is_business THEN 'business' ELSE 'freelancer' END,
  p.tier,
  p.started_at,
  'pending',  -- 가입 완료 시 'active'로
  p.id  -- 기존 partners 테이블과 매핑 유지
FROM public.partners p
JOIN public.profiles pr ON pr.name = p.name
WHERE p.status = 'active'
ON CONFLICT (profile_id) DO NOTHING;
⚠ 중요: 식별자 매핑 전략 (3가지 옵션)
  • 옵션 A · 즉시 매핑: 위 SQL처럼 임시 uuid로 profile 미리 생성. 초대 시 토큰을 그 profile에 연결. 가입 시 auth.users 생성 후 profile.id를 auth.users.id로 갱신.
  • 옵션 B · 지연 매핑 (권장): 초대 토큰만 발급, profile은 빈 상태로 안 만듦. 가입 시점에 auth.users → profile → partner_meta 한 번에 생성. legacy_partner_id로 기존 데이터 매핑.
  • 옵션 C · 매핑 안 함: 새 파트너부터만 새 시스템 사용. 기존 partners는 비모 ERP 내부 관리용으로 그대로 두고 별도 운영.

15FAQ / Edge Cases

중간에 실패하면 비모 ERP가 멈추지 않을까?
Phase 1~3은 새 테이블만 추가/백필이라 기존 시스템과 격리. Phase 4 (코드 변경)은 git revert 한 번이면 즉시 되돌아감. 가장 위험한 건 Phase 5인데, partners 백업 의무화로 최악의 경우 30분 내 복구 가능.
기존 user_profiles는 언제 지우나?
바로 안 지움. Phase 7 (수개월 후 안정 운영 확인)에서 점진 제거. role/approved 컬럼만 먼저 deprecate하고 user_profiles 자체는 남겨두는 것도 방법.
한 사람이 비모 팀 + 파트너 둘 다 해야 하면?
시스템 설계상 불가능. 그게 핵심 요구사항. 만약 정말 필요하면 별도 이메일로 두 계정을 분리 가입. 트리거가 강제로 막음.
partner_invites 토큰을 잃어버리면?
관리자가 기존 토큰을 status='revoked'로 만들고 새 토큰 발급. 일회용이지만 재발급은 자유.
RLS 무한 루프 위험은?
profiles RLS가 app_access를 참조하고 app_access RLS가 profiles를 참조하면 순환. 우리 정책은 auth.uid() 직접 비교 위주라 안전. 그래도 RLS 정책 작성 후 PostgreSQL 로그에서 stack overflow 에러 모니터링.
왜 한 번에 다 안 하고 단계로 나누나?
운영 중 시스템에 큰 변경을 한 번에 가하면 (1) 디버깅이 어렵고 (2) 롤백 단위가 너무 큼. 단계 분리는 "마지막으로 동작하던 안전 지점"을 매번 새로 만드는 것. 결국 더 빠름.
supabase/migrations/ vs Dashboard SQL Editor — 진짜 차이?
migrations 파일은 git에 남아 누가 언제 무엇을 바꿨는지 추적 가능. CI/CD에서 자동 적용도 가능. Dashboard는 빠른 일회성 작업에 좋지만 사람 기억에만 남음 — 위험. 비모 ERP는 이미 migrations 사용 중이라 일관성 위해 migrations 권장.
Supabase 무료 티어에서 PITR이 안 되는데?
in-DB 스냅샷(CREATE TABLE foo_backup AS SELECT * FROM foo)이 대안. Phase 5처럼 위험 단계 전에 무조건 스냅샷. 추가로 pg_dump로 외부 백업 받아두면 더 안전.
